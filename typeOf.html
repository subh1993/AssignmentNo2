Use of typeof() Function in JavaScript for different Data types


typeof(1)

console.log(typeof(1));

OUTPUT: "number"

typeof(1.1)

console.log(typeof(1));

OUTPUT: "number"

typeof("1.1")

console.log(typeof("1.1"));

OUTPUT: "string"

typeof(true)

console.log(typeof(true));

OUTPUT: "boolean"

typeof(null)

console.log(typeof(null));

OUTPUT: "object"

typeof(undefined)

console.log(typeof(undefined));

OUTPUT: "undefined"

typeof([])

console.log(typeof([]));

OUTPUT: "object"

typeof({})

console.log(typeof({}));

OUTPUT: "object"
